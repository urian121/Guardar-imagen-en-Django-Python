    def mi_metodo(request):
    	if request.method == 'POST':
    		form = DocumentoForm(request.POST, request.FILES)
    		if form.is_valid():
    			form.save()
    			return redirect('index')
    	else:
    		form = DocumentoForm()
    	return render(request, 'mi_template.html', {
